    deb http://s3-ap-southeast-1.amazonaws.com/thor.repository/ubuntu/dists/precise/main/binary-i386 thor contrib
  • For Ubuntu 13.10 Saucy amd64
    deb http://s3-ap-southeast-1.amazonaws.com/thor.repository/ubuntu/dists/saucy/main/binary-amd64 thor contrib
  • For Ubuntu 13.10 Saucy i386
    deb http://s3-ap-southeast-1.amazonaws.com/thor.repository/ubuntu/dists/saucy/main/binary-i386 thor contrib
